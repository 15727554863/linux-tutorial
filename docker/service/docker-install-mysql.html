          # 选择下载官方镜像

          比如，我想下载最新版本，则执行如下命令：

          docker pull mysql

          # 使用镜像

          docker run -p 3306:3306 --name mysql -v /opt/docker_v/mysql/conf:/etc/mysql/conf.d -e MYSQL_ROOT_PASSWORD=123456 -d mysql
